on_actions = {
	on_puppet = {
		effect = {
			if = {
				limit = { 
					AND = { 
						TAG = BRM
						is_subject_of = ENG
					}
				}
				set_cosmetic_tag = BRM_UK
			}
			if = {
				limit = { 
					AND = { 
						TAG = SAR
						is_subject_of = FRA
					}
				}

			}

				limit = { 
					AND = {

						is_subject_of = JAP
					}
				}

			}

				limit = {

					AND = { 
						TAG = KOR
						is_subject_of = SOV
					}
				}
				set_cosmetic_tag = KOR_SOV
			}
			if = {
				limit = {
				    AND = {
					    ROOT = {
						    original_tag = ENG
					    }
						SCO = { exists = yes }
						WLS = { exists = yes }
					}
				}
				ENG = { set_cosmetic_tag = ENG_ONLY }
			}
			if = {
				limit = {
					AND = {
					    ROOT = {
						    original_tag = CZE
					    }
						CZE = { exists = yes }
						MRV = { exists = yes }
					}
				}
				CZE = { set_cosmetic_tag = BOHEMIA } #cosmetic tag for CZE if Moravia exists
			}
		}
	}

		effect = {
			if = {
